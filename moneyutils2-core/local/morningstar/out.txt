<script type="text/javascript">
var weekDay = new Array('Sun','Mon','Tue','Wed','Thu','Fri','Sat');
function getWeekDay(){
var tradeDate = new Date("11/08/2017 12:25");
	if(tradeDate==null || typeof(tradeDate)=="undifined"){
		 $("#asOfDate").html("-")+" "+"data not available";
	}else{
		$("#asOfDate").html( weekDay[tradeDate.getDay()]+" "+formatDateByCul(tradeDate));
	}
	
}
function formatDateByCul (date){
var culture="en-US";
	if(culture==null || typeof(culture)=="undifined"){
				return this.formatDate(date,"MM/DD/yyyy HH:mm a");
			}
			if(culture.toLowerCase().replace("-","_").indexOf("fr_ca") >=0){
				return this.formatDate(date,"DD-MM-yyyy HH:mm a");
			}else if(culture.toLowerCase().replace("-","_").indexOf("fr_fr") >=0){
				return this.formatDate(date,"DD/MM/yyyy HH:mm a");
			}else if(culture.toLowerCase().replace("-","_").indexOf("en_ca") >=0){
				return this.formatDate(date,"DD/MM/yyyy HH:mm a");
			}else{
				return this.formatDate(date,"MM/DD/yyyy HH:mm a");
			}
	
	
}
function formatDate(date,formatter){
	var hour = date.getHours();
	var minute = date.getMinutes();
	var second=date.getSeconds();
	var year = date.getUTCFullYear();
	var month = date.getMonth()+1;
	var day = date.getDate();
	day=(day<10)?"0"+day:day;
	month=(month<10)?"0"+month:month;
	 var timeValue = "" +((hour >= 12) ? "PM" : "AM");
	hour = ((hour >12) ? hour -12 :hour);
	minute = (minute<10) ? "0"+minute:minute;
   
	
	return formatter.replace("MM",month).replace("yyyy",year).replace("DD",day).replace("HH",hour).replace("mm",minute).replace("a",timeValue);
}
function getCurSymbol(){
	var currency="USD";
			switch(currency.toUpperCase()){
			case "USD":
				return "$";
			case "CAD":
				return "$";
			case "GBP":
				return "￡";
			case "EUR":
				return "€";
			case "THB":
			   return "฿";
			case "CHF":
			   return "SFr.";
			case "KRW":
			   return "₩";
			default: return "&nbsp;";
			}
		}


var curSymbol=getCurSymbol();
